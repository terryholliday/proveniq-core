// What happens when getServerSession() is called

async function getServerSession(authOptions: NextAuthOptions) {
  // 1. Extract session token from cookies
  const cookies = headers().get("cookie");
  const sessionToken = parseCookies(cookies)["next-auth.session-token"];
  
  if (!sessionToken) {
    return null; // No session
  }
  
  // 2. Decode JWT (if using JWT strategy)
  try {
    const decoded = await decode({
      token: sessionToken,
      secret: process.env.NEXTAUTH_SECRET,
    });
    
    if (!decoded || decoded.exp < Date.now() / 1000) {
      return null; // Expired
    }
    
    // 3. Return session object
    return {
      user: {
        id: decoded.id,
        email: decoded.email,
        name: decoded.name,
        role: decoded.role,
      },
      expires: new Date(decoded.exp * 1000).toISOString(),
    };
  } catch {
    return null; // Invalid token
  }
}

// What happens during requirePermission()

async function requirePermission(
  organizationId: string,
  permission: Permission
): Promise<{ userId: string; organizationId: string }> {
  
  // STEP 1: Get authenticated user
  const session = await getServerSession(authOptions);
  
  if (!session?.user?.id) {
    throw new ApiError("UNAUTHORIZED", "Authentication required", 401);
  }
  
  // STEP 2: Check global role (Admin bypasses all)
  if (session.user.role === "ADMIN") {
    return { userId: session.user.id, organizationId };
  }
  
  // STEP 3: Check organization membership
  const membership = await db.organizationMember.findUnique({
    where: {
      userId_organizationId: {
        userId: session.user.id,
        organizationId,
      },
    },
  });
  
  if (!membership) {
    throw new ApiError("FORBIDDEN", "Not a member of this organization", 403);
  }
  
  // STEP 4: Check organization role permissions
  const orgPermissions = ORG_ROLE_PERMISSIONS[membership.role];
  
  if (!orgPermissions.includes(permission)) {
    // Log the denied attempt
    await audit.permissionDenied(session.user.id, organizationId, permission);
    
    throw new ApiError("FORBIDDEN", "Insufficient permissions", 403);
  }
  
  return { userId: session.user.id, organizationId };
}

## 6. RESPONSE CONSTRUCTION

### Response Pipeline

```typescript
// How responses are built

async function handleAssetCreate(req: NextRequest) {
  const startTime = Date.now();
  
  try {
    // ... business logic ...
    
    const asset = await db.asset.create({ data });
    
    // Construct success response
    return NextResponse.json(
      {
        data: asset,
        meta: {
          requestId: req.headers.get("x-request-id"),
          timestamp: new Date().toISOString(),
        },
      },
      {
        status: 201,
        headers: {
          "X-Response-Time": `${Date.now() - startTime}ms`,
          "Cache-Control": "no-store",
        },
      }
    );
    
  } catch (error) {
    // Construct error response
    return NextResponse.json(
      {
        error: {
          code: error.code || "INTERNAL_ERROR",
          message: error.message || "An error occurred",
          requestId: req.headers.get("x-request-id"),
        },
      },
      {
        status: error.status || 500,
        headers: {
          "X-Response-Time": `${Date.now() - startTime}ms`,
        },
      }
    );
  }
}
```

## 7. TRACING AND DEBUGGING

### Request Tracing

Every request should be traceable end-to-end:

```typescript
// Logging pattern for traceability

const requestId = crypto.randomUUID();

console.log(JSON.stringify({
  level: "info",
  requestId,
  stage: "middleware",
  message: "Request received",
  path: request.nextUrl.pathname,
  method: request.method,
  ip: request.ip,
  timestamp: new Date().toISOString(),
}));

// Later in the handler...
console.log(JSON.stringify({
  level: "info",
  requestId,
  stage: "database",
  message: "Query executed",
  query: "asset.create",
  duration: 25,
  timestamp: new Date().toISOString(),
}));
```

### Debug Checklist

When a request fails, check in order:

```
1. [ ] Is the request reaching the server? (Check edge logs)
2. [ ] Is authentication passing? (Check session cookie)
3. [ ] Is authorization passing? (Check membership + role)
4. [ ] Is validation passing? (Check request body)
5. [ ] Is the database query succeeding? (Check Prisma logs)
6. [ ] Is the response being constructed? (Check handler logs)
7. [ ] Is the response reaching the client? (Check network tab)
```
